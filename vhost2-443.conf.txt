<VirtualHost _default_:443>

                DocumentRoot /var/www/demo2

                ServerName demo2.com

                ServerAdmin webmaster@demo2

                ErrorLog ${APACHE_LOG_DIR}/error.log
                CustomLog ${APACHE_LOG_DIR}/access.log combined


                SSLEngine on
                SSLProtocol All -SSLv2 -SSLv3 -TLSv1 -TLSv1.1
                SSLHonorCipherOrder     on
                # <IfModule mod_headers.c>
                # Header always add Strict-Transport-Security "max-age=15768000"
                # </IfModule>
                SSLCertificateFile /etc/ssl/certs/apache-selfsigned2.crt
                SSLCertificateKeyFile /etc/ssl/private/apache-selfsigned2.key

                <Directory /var/www/demo2>
                        Require all granted
                </Directory>
</VirtualHost>
